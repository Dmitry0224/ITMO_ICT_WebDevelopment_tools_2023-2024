    Эндпойнты

    Эндпойнты в моем проекте делятся на

    • auth_endpoints - эндпойнты для работы с авторизацией
    • user_endpoints - эндпойнты для работы с пользователями
    • book_endpoints - эндпойнты для работы с книгами
    • exchange_endpoint - эндпойнты для работы с запросами на обмен

    auth_endpoints

    from fastapi import APIRouter
from auth import *
from database import get_session
from models import UserBase, User, UserShow, ChangePassword, UserCreate
from sqlmodel import select


auth_router = APIRouter()


@auth_router.post('/registration', status_code=201, description='Register new user')
def register(user: UserCreate, session=Depends(get_session)):
    users = session.exec(select(User)).all()
    if any(x.username == user.username for x in users):
        raise HTTPException(status_code=400, detail='Username is taken')
    hashed_pwd = get_password_hash(user.password)
    user = User(username=user.username, password=hashed_pwd)
    session.add(user)
    session.commit()
    return {"status": 201, "message": "Created"}


@auth_router.post('/login')
def login(user: UserCreate, session=Depends(get_session)):
    user_found = session.exec(select(User).where(User.username == user.username)).first()
    if not user_found:
        raise HTTPException(status_code=401, detail='Invalid username and/or password')
    verified = verify_password(user.password, user_found.password)
    if not verified:
        raise HTTPException(status_code=401, detail='Invalid username and/or password')
    token = encode_token(user_found.username)
    return {'token': token}


@auth_router.get('/me', response_model=UserShow)
def get_current_user(user: User = Depends(get_current_user)) -> User:
    return user


@auth_router.patch("/me/change-password")
def user_pwd(user_pwd: ChangePassword, session=Depends(get_session), current=Depends(get_current_user)):
    found_user = session.get(User, current.id)
    if not found_user:
        raise HTTPException(status_code=404, detail="User not found")
    verified = verify_password(user_pwd.old_password, found_user.password)
    if not verified:
        raise HTTPException(status_code=400, detail="Invalid old password")
    hashed_pwd = get_password_hash(user_pwd.new_password)
    found_user.password = hashed_pwd
    session.add(found_user)
    session.commit()
    session.refresh(found_user)
    return {"status": 200, "message": "password changed successfully"}

    book_endpoints

    from fastapi import APIRouter, HTTPException, Depends
from typing_extensions import TypedDict
from models import BookCreate, Book, BookShow, User
from database import get_session
from auth import get_current_user
book_router = APIRouter()

@book_router.post("/create")
def create_book(book: BookCreate, user: User = Depends(get_current_user), session=Depends(get_session)) -> TypedDict('Response', {"status": int, "data": Book}):
    book_data = book.dict()
    book_data["owner_id"] = user.id
    new_book = Book(**book_data)
    session.add(new_book)
    session.commit()
    session.refresh(new_book)
    return {"status": 200, "data": new_book}

@book_router.get("/list")
def list_books(session=Depends(get_session)) -> list[BookShow]:
    return session.query(Book).all()

@book_router.get("/{book_id}", response_model=BookShow)
def get_book(book_id: int, session=Depends(get_session)):
    obj = session.get(Book, book_id)
    if obj is None:
        raise HTTPException(status_code=404, detail="Book not found")
    return obj

@book_router.patch("/update/{book_id}")
def update_book(book_id: int, book: BookCreate, session=Depends(get_session)) -> Book:
    db_book = session.get(Book, book_id)
    if not db_book:
        raise HTTPException(status_code=404, detail="Book not found")

    book_data = book.dict(exclude_unset=True)
    for key, value in book_data.items():
        setattr(db_book, key, value)
    session.add(db_book)
    session.commit()
    session.refresh(db_book)
    return db_book

@book_router.delete("/delete/{book_id}")
def delete_book(book_id: int, session=Depends(get_session)):
    book = session.get(Book, book_id)
    if not book:
        raise HTTPException(status_code=404, detail="Book not found")
    session.delete(book)
    session.commit()
    return {"ok": True}

    exchange_endpoint

    from fastapi import APIRouter, HTTPException, Depends
from models import ExchangeRequestBase, ExchangeRequest, ExchangeRequestShow, User
from database import get_session
from typing_extensions import TypedDict
from auth import get_current_user

exchange_request_router = APIRouter()

@exchange_request_router.post("/create/{book_id}")
def exchange_request_create(
    book_id: int,
    exchange_request: ExchangeRequestBase,
    current_user: User = Depends(get_current_user),
    session=Depends(get_session)
) -> TypedDict('Response', {"status": int, "data": ExchangeRequest}):
    exchange_request_data = exchange_request.dict()
    exchange_request_data["book_id"] = book_id
    exchange_request_data["requester_id"] = current_user.id

    new_exchange_request = ExchangeRequest(**exchange_request_data)
    session.add(new_exchange_request)
    session.commit()
    session.refresh(new_exchange_request)
    return {"status": 200, "data": new_exchange_request}

@exchange_request_router.get("/list")
def exchange_request_list(session=Depends(get_session)) -> list[ExchangeRequest]:
    return session.query(ExchangeRequest).all()


@exchange_request_router.get("/{request_id}", response_model=ExchangeRequestShow)
def exchange_request_get(request_id: int, session=Depends(get_session)):
    obj = session.get(ExchangeRequest, request_id)
    if obj is None:
        raise HTTPException(status_code=404, detail="Exchange request not found")
    return obj


@exchange_request_router.patch("/update/{request_id}")
def exchange_request_update(request_id: int, exchange_request: ExchangeRequestBase, session=Depends(get_session)) -> ExchangeRequest:
    db_request = session.get(ExchangeRequest, request_id)
    if not db_request:
        raise HTTPException(status_code=404, detail="Exchange request not found")

    request_data = exchange_request.model_dump(exclude_unset=True)
    for key, value in request_data.items():
        setattr(db_request, key, value)
    session.add(db_request)
    session.commit()
    session.refresh(db_request)
    return db_request


@exchange_request_router.delete("/delete/{request_id}")
def exchange_request_delete(request_id: int, session=Depends(get_session)):
    exchange_request = session.get(ExchangeRequest, request_id)
    if not exchange_request:
        raise HTTPException(status_code=404, detail="Exchange request not found")
    session.delete(exchange_request)
    session.commit()
    return {"ok": True}
