    Threads vs Porcess vs Async

    Сравнение перфоманса 3 подходов в python:

    • multithread
    • multiprocess
    • async

    Код программы

    import asyncio
import threading
import multiprocessing
import time


def parts_thread_and_mult(s, e, res, idx):
    res[idx] = sum(range(s, e + 1))


def sum_threading():
    num_tasks = 4
    n = 1000000
    tasks_list = []
    result = [0] * num_tasks
    step = n // num_tasks

    for i in range(num_tasks):
        start = i * step + 1
        end = (i + 1) * step if i != num_tasks - 1 else n
        thread = threading.Thread(target=parts_thread_and_mult, args=(start, end, result, i))
        tasks_list.append(thread)
        thread.start()

    for thread in tasks_list:
        thread.join()

    total = sum(result)
    return total



def sum_multi():
    num_tasks = 4
    n = 1000000
    tasks_list = []
    step = n // num_tasks

    manager = multiprocessing.Manager()
    res = manager.list([0] * num_tasks)

    for i in range(num_tasks):
        s = i * step + 1
        e = (i + 1) * step if i != num_tasks - 1 else n
        process = multiprocessing.Process(target=parts_thread_and_mult, args=(s, e, res, i))
        tasks_list.append(process)
        process.start()

    for process in tasks_list:
        process.join()

    _sum = sum(res)
    return _sum


async def parts_async(s, e):
    return sum(range(s, e + 1))


async def sum_async():
    num_tasks = 4
    n = 1000000
    step = n // num_tasks
    tasks = []

    for i in range(num_tasks):
        s = i * step + 1
        e = (i + 1) * step if i != num_tasks - 1 else n
        tasks.append(asyncio.create_task(parts_async(s, e)))

    results = await asyncio.gather(*tasks)
    total = sum(results)
    return total

if __name__ == '__main__':
    start_time = time.time()
    total = asyncio.run(sum_async())
    end_time = time.time()
    print(f"Async time: {end_time - start_time} seconds")

    start_time = time.time()
    total = sum_multi()
    end_time = time.time()
    print(f"Multiprocessing time: {end_time - start_time} seconds")


    start_time = time.time()
    total = sum_threading()
    end_time = time.time()
    print(f"Threading time: {end_time - start_time} seconds")

    Результаты

    Async time: 0.0416417121887207 seconds
Multiprocessing time: 0.6034059524536133 seconds
Threading time: 0.0812838077545166 seconds

    Выводы

    В данном задании лучше всего себя показал async, чуть хуже multiprocessing, ну и хуже всего threading, так как затрачивает больше всего ресурсов на создание потоков
